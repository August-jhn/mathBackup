\fbox{definition} We define $\cos z = \frac{e^{iz} + e^{-iz}}{2}$ and $\sin z = \frac{e^{iz} + e^{-iz}}{2i}$.\\

\fbox{theorem} $\deriv{\sin z} = \deriv{\frac{e^{iz} + e^{-iz}}{2i}} = \cos z$. So this is what we'd want, likeiwse with $\deriv{\cos{z}}$\\


\fbox{definitoin} $\sinh{y} = \frac{e^y-e^{-y}}{2}$ and $\cosh{y} = \frac{e^y + e^{-y}}{2}$. 
\\
\fbox{theorem} To find $\deriv\sinh{z} = \deriv{\frac{e^z - e^{-z}}{2}} = \frac{e^z+ e^{-z}}{2} = \cosh{z}$.\\

\fbox{theorem} To find $\deriv \cosh{z} = \deriv{\frac{e^z + e^{-z}}{2}} = \sinh{z}$.\\

\fbox{theorem} $\sinh{-z} = -\sinh{z}$ and $\cosh{-z} = \cosh{z}$.\\

\fbox{theorem} $\sinh^2{z} - \cosh^2{z} = 1$.\\

\fbox{theorme} $\sinh(z_1 + z_2) = -i\sin(iz_1 + iz_2) = -i(\sin(iz_1)\cos(iz_2) + \cos(iz_1)\sin(iz_2)) = \sinh(z_1)\cosh(z_2) + \cosh(z_1)\sinh(z_2)$.\\

\fbox{definition} $\tanh{z} = \frac{\sinh z}{\cosh z}$.\\

\fbox{definition} $\arcsin{z} = -i\log(iz + (1-z^2)^{1/2})$.\\
